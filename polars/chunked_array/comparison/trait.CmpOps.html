Compare Series

and ChunkedArray's and get a boolean mask that can be use to filter rows.

Example

use polars::prelude::*;
fn filter_all_ones(df: &DataFrame) -> Result<DataFrame> {
    let mask = df
    .column("column_a")
    .ok_or(PolarsError::NotFound)?
    .eq(1);

    df.filter(&mask)
}
